import image1 from "../../../assets/img.png"; // Import your local images
import image2 from "../../../assets/img2.png";
import image3 from "../../../assets/img3.png";

const Home = () => {
  const slides = [
    {
      image: image3,
      region: "Seul",
      type: "Mosque",
      location: "118.46.303.1",
    },
    {
      image: image2,
      region: "Tokyo",
      type: "Shop",
      location: "35.6895° N",
    },

      image: image1,
      region: "Istanbul",
      type: "Restaurant",
      location: "41.0082° N, ",
    },
  ];
